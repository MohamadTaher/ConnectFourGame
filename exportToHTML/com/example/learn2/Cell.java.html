package com.example.learn2;

import android.graphics.Color;
import android.graphics.Paint;
import android.graphics.RectF;

public class Cell {

    private final int row;
    private final int col;

    private final int cellSize;
    private boolean occupied;
    private final int left;
    private final int top;

    private int color;
    private RectF cellShape;
    private RectF circleShape;

    public Cell(int row, int col, int left, int top, int cellSize) {
        this.row = row;
        this.col = col;
        this.left = left;
        this.top = top;
        this.cellSize = cellSize;
        this.occupied = false;
        this.color = Color.WHITE; // default color is white

        cellShape = new RectF(left, top, left + cellSize, top + cellSize);
        circleShape = new RectF(left + cellSize / 4, top + cellSize / 4, left + 3 * cellSize / 4, top + 3 * cellSize / 4);

    }
    public RectF getCircleShape() {
        return circleShape;
    }

    public Paint getCirclePaint() {
        Paint paint = new Paint();
        paint.setColor(Color.RED);
        paint.setStyle(Paint.Style.FILL);
        return paint;
    }

    public int getCenter(){
        return cellSize / 2;
    }

    public int getCellSize() {
        return cellSize;
    }

    public int getRow() {
        return row;
    }

    public int getCol() {
        return col;
    }

    public boolean isOccupied() {
        return occupied;
    }

    public void setOccupied(boolean occupied) {
        this.occupied = occupied;
    }

    public int getColor() {
        return color;
    }

    public void setColor(int color) {
        this.color = color;
    }

    public int getLeft() {
        return left;
    }

    public int getTop() {
        return top;
    }

    public int getRight() {
        return left + cellSize;
    }

    public int getBottom() {
        return top + cellSize;
    }

}
